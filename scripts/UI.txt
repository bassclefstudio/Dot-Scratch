Define Build Text Segments (warp=true)
{
    render.c1 = 0;
    List.DeleteAll(textSegments);
    Repeat (List.Length(SegmentType))
    {
        render.c1 += 1;
        If (SegmentType[render.c1] == "UI")
        {
            List.Add(textSegments, render.c1);
        }
    }
}

Define Clear | Character Data (warp=true)
{
    Looks.SwitchCostumeTo(blank);
    Looks.SwitchCostumeTo((Costume.GetNumberName(number) - 1));
    If ((Costume.GetNumberName(number) - 1) > List.Length(_chIndex))
    {
        _i0 = (Costume.GetNumberName(number) - 1);
    }
    Else
    {
        _i0 = List.Length(_chIndex);
    }
    List.DeleteAll(_chIndex);
    List.DeleteAll(_chWidth);
    List.DeleteAll(_chData0);
    List.DeleteAll(_chData1);
    List.DeleteAll(_chData2);
    List.DeleteAll(_chData3);
    Repeat (_i0)
    {
        List.Add(_chData0, "__");
        List.Add(_chData1, "none");
        List.Add(_chData2, "");
        List.Add(_chData3, "");
        List.Add(_chIndex, (List.Length(_chData0) + 1));
        List.Add(_chWidth, (1 * 0));
        Repeat (25)
        {
            List.Add(_chData0, "");
            List.Add(_chData1, "");
            List.Add(_chData2, "");
            List.Add(_chData3, "");
        }
    }
    List.Add(_chData0, "__");
    List.Add(_chData1, "none");
    List.Add(_chData2, "");
    List.Add(_chData3, "");
}
